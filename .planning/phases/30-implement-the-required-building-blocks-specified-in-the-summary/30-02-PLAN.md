```rust
use crate::types::ability::ResolvedAbility;
use crate::types::events::GameEvent;
use crate::types::game_state::GameState;
use crate::types::ability::Effect;

pub fn resolve(
    state: &mut GameState,
    ability: &ResolvedAbility,
    events: &mut Vec<GameEvent>,
) {
    if let Effect::AddRestriction { restriction } = &ability.effect {
        // CR 614.16: Add restriction to game state
        state.restrictions.push(restriction.clone());
        events.push(GameEvent::EffectResolved {
            effect: crate::types::ability::effect_variant_name(&ability.effect).to_string(),
        });
    }
}
```

**In `crates/engine/src/game/replacement.rs`:**

1. Add a helper function `is_prevention_disabled()`:
   ```rust
   /// CR 614.16: Check if damage prevention is disabled by a GameRestriction.
   fn is_prevention_disabled(state: &GameState, proposed: &ProposedEvent) -> bool {
       state.restrictions.iter().any(|r| match r {
           GameRestriction::DamagePreventionDisabled { scope, .. } => {
               match scope {
                   None => true, // Global — all prevention disabled
                   Some(RestrictionScope::SpecificSource(id)) => {
                       matches!(proposed, ProposedEvent::Damage { source, .. } if source == id)
                   }
                   Some(RestrictionScope::SourcesControlledBy(pid)) => {
                       if let ProposedEvent::Damage { source, .. } = proposed {
                           state.objects.get(source)
                               .map(|obj| obj.controller == *pid)
                               .unwrap_or(false)
                       } else { false }
                   }
                   Some(RestrictionScope::DamageToTarget(tid)) => {
                       matches!(proposed, ProposedEvent::Damage { target, .. } if target == tid)
                   }
               }
           }
       })
   }
   ```
