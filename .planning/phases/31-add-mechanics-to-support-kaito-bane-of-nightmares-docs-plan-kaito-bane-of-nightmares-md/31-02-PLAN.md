   ```rust
   pub fn resolve_quantity(
       state: &GameState,
       expr: &QuantityExpr,
       controller: PlayerId,
       source_id: ObjectId,
   ) -> i32 {
       match expr {
           QuantityExpr::Fixed { value } => *value,
           QuantityExpr::Ref { qty } => match qty {
               QuantityRef::HandSize => { /* existing */ },
               QuantityRef::LifeTotal => { /* existing */ },
               QuantityRef::GraveyardSize => { /* existing */ },
               QuantityRef::LifeAboveStarting => { /* existing */ },
               QuantityRef::ObjectCount { filter } => {
                   state.battlefield.iter()
                       .filter(|&&id| filter_matches_object(state, id, filter, controller))
                       .count() as i32
               }
               QuantityRef::PlayerCount { filter } => {
                   resolve_player_count(state, filter, controller)
               }
               QuantityRef::CountersOnSelf { counter_type } => {
                   state.objects.get(&source_id)
                       .map(|obj| {
                           let ct = parse_counter_type(counter_type);
                           obj.counters.get(&ct).copied().unwrap_or(0) as i32
                       })
                       .unwrap_or(0)
               }
           },
       }
   }
   ```
   Where `resolve_player_count` iterates opponents checking the PlayerFilter condition.
   Use the existing `filter_matches` infrastructure from `game/filter.rs` for ObjectCount matching.
