1. In `crates/engine/src/types/ability.rs`, add three new variants to `StaticCondition`:
   ```rust
   And {
       conditions: Vec<StaticCondition>,
   },
   Or {
       conditions: Vec<StaticCondition>,
   },
   HasCounters {
       counter_type: String,
       minimum: u32,
   },
   ```
   Use `String` for counter_type to match existing `CounterType::Generic(String)` pattern and serde compat. The `HasCounters` variant is self-referential — it checks the object the static belongs to.

2. In `crates/engine/src/game/layers.rs`, find the `evaluate_condition()` function and add match arms:
   ```rust
   StaticCondition::And { conditions } => {
       conditions.iter().all(|c| evaluate_condition(state, c, controller, source_id))
   }
   StaticCondition::Or { conditions } => {
       conditions.iter().any(|c| evaluate_condition(state, c, controller, source_id))
   }
   StaticCondition::HasCounters { counter_type, minimum } => {
       // CR 114.3: Check counters on the source object
       state.objects.get(&source_id)
           .map(|obj| {
               let count = match counter_type.as_str() {
                   "+1/+1" | "plus1plus1" => obj.counters.get(&CounterType::Plus1Plus1).copied().unwrap_or(0),
                   "-1/-1" | "minus1minus1" => obj.counters.get(&CounterType::Minus1Minus1).copied().unwrap_or(0),
                   "loyalty" => obj.counters.get(&CounterType::Loyalty).copied().unwrap_or(0),
                   "stun" => obj.counters.get(&CounterType::Stun).copied().unwrap_or(0),
                   other => obj.counters.get(&CounterType::Generic(other.to_string())).copied().unwrap_or(0),
               };
               count >= *minimum
           })
           .unwrap_or(false)
   }
   ```
   Note: `evaluate_condition` already takes `source_id` — confirm this by reading the function signature. If it only takes `(state, condition, controller)`, add `source_id: ObjectId` parameter and update all call sites.

No new code needed for the animation itself — the compound condition + existing modification application is sufficient.
